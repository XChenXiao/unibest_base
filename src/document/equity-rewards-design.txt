<!-- 理财应用股权奖励组件 -->
<!-- components/equity-rewards.vue -->
<template>
  <view class="equity-rewards-container">
    <!-- 奖励模块标题 -->
    <view class="rewards-title">
      <text class="title-text">股权奖励</text>
    </view>
    
    <!-- 奖励卡片列表 -->
    <view class="rewards-list">
      <!-- 注册实名奖励卡片 -->
      <view class="reward-card">
        <view class="reward-icon-container">
          <view class="reward-icon register-icon">
            <text class="uni-icons uniui-person-filled"></text>
          </view>
        </view>
        <view class="reward-content">
          <view class="reward-header">
            <text class="reward-name">注册实名奖励</text>
            <text class="reward-amount">{{ registerReward }}股</text>
          </view>
          <text class="reward-desc">完成注册并实名认证可获得股权奖励</text>
          <button 
            class="reward-btn" 
            :class="{'claimed-btn': isRegisterRewardClaimed}"
            :disabled="isRegisterRewardClaimed" 
            @click="claimReward('register')"
          >
            {{ isRegisterRewardClaimed ? '已领取' : '领取奖励' }}
          </button>
        </view>
      </view>
      
      <!-- 邀请注册实名奖励卡片 -->
      <view class="reward-card">
        <view class="reward-icon-container">
          <view class="reward-icon invite-icon">
            <text class="uni-icons uniui-upload"></text>
          </view>
        </view>
        <view class="reward-content">
          <view class="reward-header">
            <text class="reward-name">邀请注册实名奖励</text>
            <text class="reward-amount">{{ inviteReward }}股</text>
          </view>
          <text class="reward-desc">邀请好友注册并完成实名认证</text>
          <!-- 进度条 -->
          <view class="progress-container">
            <view class="progress-bar">
              <view 
                class="progress-filled" 
                :style="{ width: `${(inviteProgress / inviteTarget) * 100}%` }"
              ></view>
            </view>
            <text class="progress-text">{{ inviteProgress }}/{{ inviteTarget }}</text>
          </view>
          <button 
            class="reward-btn" 
            :class="{'disabled-btn': inviteProgress < inviteTarget}" 
            :disabled="inviteProgress < inviteTarget"
            @click="claimReward('invite')"
          >
            {{ inviteProgress < inviteTarget ? `还需邀请${inviteTarget - inviteProgress}人` : '领取奖励' }}
          </button>
        </view>
      </view>
    </view>
  </view>
</template>

<script lang="ts" setup>
import { defineProps, defineEmits } from 'vue';

// 定义属性
const props = defineProps({
  registerReward: {
    type: Number,
    default: 50
  },
  inviteReward: {
    type: Number,
    default: 100
  },
  isRegisterRewardClaimed: {
    type: Boolean,
    default: false
  },
  inviteProgress: {
    type: Number,
    default: 0
  },
  inviteTarget: {
    type: Number,
    default: 5
  }
});

// 定义事件
const emit = defineEmits(['claim-reward']);

// 领取奖励处理函数
const claimReward = (type: 'register' | 'invite') => {
  if (type === 'register' && props.isRegisterRewardClaimed) {
    return;
  }
  
  if (type === 'invite' && props.inviteProgress < props.inviteTarget) {
    return;
  }
  
  // 触发领取奖励事件
  emit('claim-reward', type);
};
</script>
